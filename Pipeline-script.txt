pipeline{

	agent any

	environment {
		DOCKERHUB_CREDENTIALS=credentials('abinaya251')
	}

	stages {
	    
	    stage('gitclone') {

			steps {
				git 'https://github.com/Abinaya-251/SimpleFlaskUI.git'
			}
		}

		stage('Build') {

			steps {
				sh 'docker build -t abinaya251/python-jenkins .'
			}
		}

		stage('Login') {

			steps {
				sh 'echo $DOCKERHUB_CREDENTIALS_PSW | docker login -u $DOCKERHUB_CREDENTIALS_USR --password-stdin'
			}
		}

		stage('Push') {

			steps {
				sh 'docker push abinaya251/python-jenkins:latest'
			}
		}
	    stage('Deploy to Test Server') {
	        
            steps {
                script {
                def dockerRun = 'docker run -d --name my-app -p 5003:5003 abinaya251/python-jenkins'
                sshagent(['ssh-id']) {
                // some block
                sh "ssh -o StrictHostKeyChecking=no ubuntu@44.203.126.107 ${dockerRun}"
            }
        }
    }
    }
	}
}
